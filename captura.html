const escuelas = JSON.parse(rawData); let coincidenciaEncontrada = null; let escuelaMatch = null; // Normalizar texto de búsqueda eliminando acentos redundantes const normalizar = (txt) => txt.normalize("NFD").replace(/[\u0300-\u036f]/g, "").toUpperCase().trim(); const busquedaNorm = normalizar(nombreABuscar); // Recorrer el árbol JSON estructurado for (const esc of escuelas) { const benEncontrado = esc.beneficiarios.find(b => normalizar(b.n) === busquedaNorm || (curp && curp.length === 18 && busquedaNorm.includes(normalizar(b.n)))); if (benEncontrado) { coincidenciaEncontrada = benEncontrado; escuelaMatch = esc; break; } } const cardCoincidencia = document.getElementById('coincidencia-exacta-card'); const cardError = document.getElementById('match-no-encontrado-card'); if (coincidenciaEncontrada) { // Ocultar error y mostrar datos estructurados cardError.style.display = 'none'; cardCoincidencia.classList.add('visible'); // Inyectar datos en la mini tabla de coincidencia document.getElementById('match-nombre').textContent = coincidenciaEncontrada.n; document.getElementById('match-curp-lbl').textContent = curp.length === 18 ? `CURP Validada: ${curp}` : 'Verificado por Nombre Completo'; document.getElementById('match-escuela').textContent = escuelaMatch.nombre; document.getElementById('match-cct').textContent = escuelaMatch.cct; // Nivel Educativo const badgeNivel = document.getElementById('match-nivel-badge'); badgeNivel.textContent = escuelaMatch.nivel; if(escuelaMatch.nivel === "PRIMARIA") { badgeNivel.className = "nivel-badge pri"; } else { badgeNivel.className = "nivel-badge sec"; } // Programa y fecha document.getElementById('match-programa').textContent = escuelaMatch.programa; document.getElementById('match-fecha').textContent = escuelaMatch.fecha; document.getElementById('scan-status-text').innerHTML = "✓ ¡Coincidencia localizada con éxito!"; } else { // Mostrar bloque de no encontrado cardCoincidencia.classList.remove('visible'); cardError.style.display = 'block'; document.getElementById('scan-status-text').innerHTML = "× No se localizó ningún registro coincidente."; } } function buscarManualmente() { const val = document.getElementById('input-search').value; if (val.length > 4) { // Reutiliza la función pasándole el valor manual compararContraNominal(val, ""); } } function limpiarFormularioScanner() { document.getElementById('input-search').value = ""; document.getElementById('alerta-curp-contenedor').style.display = 'none'; document.getElementById('coincidencia-exacta-card').classList.remove('visible');
